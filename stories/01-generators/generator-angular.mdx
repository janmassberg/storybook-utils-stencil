import { Meta, Source } from "@storybook/blocks";
import {
    generateSourceCodeAngularComponent,
    generateSourceCodeAngularHtml,
} from "../../src";
import { testArgsButton } from "../../src/test-utils/testArgs";

export const usageExample = `
import {
    generateSourceCodeAngularComponent,
    generateSourceCodeAngularHtml,
} from "@jmassberg/storybook-utils";\n
const args = {
    required: true,
    variant: "primary",
    onClick: (event) => {
        // Handle click event
        console.log(event.target);
    },
    _slot: "Label",
};\n
const angularHtmlCode = generateSourceCodeAngularHtml("ui-button", args);
const angularComponentCode = generateSourceCodeAngularComponent("ui-button", args);
`;

<Meta title="Code Generators/Angular" />
